Tworzymy tablice: cursor i note, które przechowywują pozycje.

Nasłuchujemy zdarzenia mousedown i sprawdzamy, czy kliknięty element ma klasę notka.

if(e.target.classList.contains("notka")){

Jeśli tak, zapisujemy pozycję kursora oraz pozycję elementu na ekranie za pomocą getBoundingClientRect().

Podczas mousemove sprawdzamy, czy istnieje aktywny przeciągany element.

Jeśli tak, obliczamy różnicę między aktualną pozycją kursora a miejscem rozpoczęcia przeciągania.

let odleglosc = {
x: obecnyCursor.x - cursor.x,
y: obecnyCursor.y - cursor.y
}

Na tej podstawie aktualizujemy pozycję elementu, ustawiając left i top oraz width i height przy zmianie rozmiaru.

Zmieniamy kursor na grab lub nwse-resize, aby pokazać tryb przeciągania.

Podczas mouseup kończymy przeciąganie i resetujemy stan oraz kursor.

Eventlistery wszystkie:

document.addEventListener("mousedown", (e) => {
    if(e.target.classList.contains("notka") || e.target.classList.contains('resize')){
        e.preventDefault()
        cursor = {
            x: e.clientX,
            y: e.clientY
        }
        if (e.target.classList.contains('resize')){
            note = {
                dom: e.target.closest(".notka"),
                x: e.target.closest(".notka").getBoundingClientRect().width,
                y: e.target.closest(".notka").getBoundingClientRect().height,
                resize: true
            }
        }else{
            note = {
                dom: e.target.closest(".notka"),
                x: e.target.getBoundingClientRect().left,
                y: e.target.getBoundingClientRect().top,
                resize: false
            }
        }
    }
})

document.addEventListener("mousemove", (e) => {
    if(note.dom == null) return;

    let obecnyCursor = {
        x: e.clientX,
        y: e.clientY
    }

    let odleglosc = {
        x: obecnyCursor.x - cursor.x,
        y: obecnyCursor.y - cursor.y
    }

    if(note.resize){
        note.dom.style.width = (note.x + odleglosc.x) + "px"
        note.dom.style.height = (note.y + odleglosc.y) + "px"
        note.dom.style.cursor = "nwse-resize"
    }else{
        note.dom.style.left = (note.x + odleglosc.x) + "px"
        note.dom.style.top = (note.y + odleglosc.y) + "px"
        note.dom.style.cursor = "grab"
    }
})

document.addEventListener("mouseup", (e) => {
    if(note.dom == null) return;

    note.dom.style.cursor = "auto";
    note.dom = null
})
